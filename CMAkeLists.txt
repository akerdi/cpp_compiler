cmake_minimum_required(VERSION 2.9)
project(app)

set(COMPILER_VERSION_MAJOR 1)
set(COMPILER_VERSION_MINOR 0)

option(USE_MY_COMPILER
    "Use Provided compiler implementation" OFF)

configure_file(
    "${PROJECT_SOURCE_DIR}/config.h.in"
    "${PROJECT_SOURCE_DIR}/config.h"
)

aux_source_directory(. DIR_SRCS)

if (USE_MY_COMPILER)
    message("use my compiler on")
    add_library(COMPILER compiler.cpp)
else()
    message("use my compiler offffff")
    add_executable(app ${DIR_SRCS})
endif()
